    The module structure:

    ```javascript
    import { dbOps } from '../db/operations.js';
    import { calculateMatchDelta, INITIAL_ELO } from './elo.js';

    let reconciling = false;
    let recomputeInProgress = false;

    export function isReconciling() { return reconciling; }
    export function isRecomputeInProgress() { return recomputeInProgress; }
    export function setRecomputeInProgress(value) { recomputeInProgress = value; }

    export async function runReconciliation() {
      if (reconciling) return;        // fast skip — MUST be before any await
      reconciling = true;             // set synchronously
      try {
        const now = Date.now();
        const pendingMatches = await dbOps.getPendingMatches();
        for (const pm of pendingMatches) {
          try {
            // ... per-match expiry logic (lines 15-68 of original state.js) ...
          } catch (err) {
            console.error(`[reconciliation] skipping pending match ${pm.id}:`, err);
            // continue loop — match stays pending for retry on next call
          }
        }
        // cleanup loop: delete confirmed/rejected pending matches
        for (const pm of pendingMatches) {
          if (pm.status === 'confirmed' || pm.status === 'rejected') {
            try { await dbOps.deletePendingMatch(pm.id); } catch (e) { console.error(e); }
          }
        }
      } finally {
        reconciling = false;
      }
    }
    ```

    CRITICAL ordering: `if (reconciling) return` then `reconciling = true` — both before any `await`. This prevents re-entrant execution even across event loop ticks.

    import { Router } from 'express';
    import { dbOps } from '../db/operations.js';
    import { authMiddleware } from '../middleware/auth.js';
    import { runReconciliation, isRecomputeInProgress } from '../services/reconciliation.js';

    const router = Router();

    router.get('/state', authMiddleware, async (req, res) => {
      try {
        await runReconciliation();
        const state = await dbOps.getFullState({
          matchesLimit: 100,
          historyLimit: 200,
          challengesLimit: 50,
          tournamentsLimit: 20,
        });
        state.pendingMatches = state.pendingMatches.filter(
          (pm) => pm.status === 'pending' || pm.status === 'disputed'
        );
        state.recomputeInProgress = isRecomputeInProgress();
        res.json(state);
      } catch (err) {
        console.error('Error in /api/state:', err);
        res.status(500).json({ error: 'Internal server error' });
      }
    });

    export default router;
