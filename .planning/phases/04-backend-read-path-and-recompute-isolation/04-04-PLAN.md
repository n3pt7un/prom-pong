    STEP A — Add runRecompute() to source/server/services/reconciliation.js.

    Append to the existing exports (do NOT remove or change runReconciliation, isReconciling, isRecomputeInProgress, setRecomputeInProgress):

    ```javascript
    export async function runRecompute(dbOps) {
      if (recomputeInProgress) throw new Error('Recompute already in progress');
      recomputeInProgress = true;   // set synchronously before first await
      try {
        return await dbOps.recalculateElo();
      } finally {
        recomputeInProgress = false;
      }
    }
    ```

    const CHUNK_SIZE = 50;
    for (let i = 0; i < sortedMatches.length; i += CHUNK_SIZE) {

        // ... per-match ELO logic — IDENTICAL to current, no changes ...
      }
      if (i + CHUNK_SIZE < sortedMatches.length) {

    }
