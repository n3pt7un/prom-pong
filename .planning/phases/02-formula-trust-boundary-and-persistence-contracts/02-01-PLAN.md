/**
 * SECU-01: Formula trust boundary — admin-only gate on ELO config endpoints.
 * SECU-02: Formula compatibility — known preset outputs preserved after Phase 2 changes.
 *
 * Test strategy:
 * - Auth/admin boundary: inline boundary logic (Phase 1 pattern) — no firebase-admin.
 * - Compatibility: direct import of calculateMatchDelta from elo.js.
 */

import { calculateMatchDelta, validateCustomFormula, FORMULA_PRESETS } from '../services/elo.js';

// ─── Helpers (Phase 1 pattern) ─────────────────────────────────────────────────

function mockRPC(headers = {}, user = null) {
  const req = { headers, user };
  const res = {
    _status: null,
    _body: null,
    status(code) { this._status = code; return this; },
    json(data) { this._body = data; return this; },
  };
  let nextCalled = false;
  return { req, res, next: () => { nextCalled = true; }, isNextCalled: () => nextCalled };
}

async function adminBoundaryCheck(req, res, next, admins) {
  if (!req.user) return res.status(401).json({ error: 'Authentication required' });
  const isAdmin = admins.some(a => a.firebaseUid === req.user.uid);
  if (!isAdmin) return res.status(403).json({ error: 'Admin access required' });
  next();
}

// ─── SECU-01: Admin-only gate on ELO config endpoints ─────────────────────────

describe('SECU-01: formula auth boundary — GET /admin/elo-config', () => {
  const admins = [{ firebaseUid: 'admin-uid-1' }];

  test('unauthenticated (no req.user) → 401', async () => {
    const { req, res, next, isNextCalled } = mockRPC();
    req.user = null;
    await adminBoundaryCheck(req, res, next, admins);
    expect(res._status).toBe(401);
    expect(res._body).toMatchObject({ error: 'Authentication required' });
    expect(isNextCalled()).toBe(false);
  });

  test('authenticated non-admin → 403', async () => {
    const { req, res, next, isNextCalled } = mockRPC();
    req.user = { uid: 'non-admin-uid', email: 'user@example.com' };
    await adminBoundaryCheck(req, res, next, admins);
    expect(res._status).toBe(403);
    expect(res._body).toMatchObject({ error: 'Admin access required' });
    expect(isNextCalled()).toBe(false);
  });

  test('authenticated admin → calls next()', async () => {
    const { req, res, next, isNextCalled } = mockRPC();
    req.user = { uid: 'admin-uid-1', email: 'admin@example.com' };
    await adminBoundaryCheck(req, res, next, admins);
    expect(isNextCalled()).toBe(true);
    expect(res._status).toBeNull();
  });
});

describe('SECU-01: formula auth boundary — PUT /admin/elo-config', () => {
  const admins = [{ firebaseUid: 'admin-uid-1' }];

  test('unauthenticated → 401', async () => {
    const { req, res, next, isNextCalled } = mockRPC();
    req.user = null;
    await adminBoundaryCheck(req, res, next, admins);
    expect(res._status).toBe(401);
    expect(isNextCalled()).toBe(false);
  });

  test('authenticated non-admin → 403', async () => {
    const { req, res, next, isNextCalled } = mockRPC();
    req.user = { uid: 'other-uid', email: 'other@example.com' };
    await adminBoundaryCheck(req, res, next, admins);
    expect(res._status).toBe(403);
    expect(isNextCalled()).toBe(false);
  });

  test('authenticated admin → calls next()', async () => {
    const { req, res, next, isNextCalled } = mockRPC();
    req.user = { uid: 'admin-uid-1', email: 'admin@example.com' };
    await adminBoundaryCheck(req, res, next, admins);
    expect(isNextCalled()).toBe(true);
  });
});

// ─── SECU-02: Formula preset compatibility — known outputs ────────────────────

describe('SECU-02: standard preset — known ELO delta outputs', () => {
  const cfg = { formulaPreset: 'standard', kFactor: 32, dFactor: 200 };

  test('equal ratings 1200 vs 1200 → delta 16', () => {
    expect(calculateMatchDelta(1200, 1200, cfg)).toBe(16);
  });

  test('favourite wins 1400 vs 1200 → delta 3', () => {
    expect(calculateMatchDelta(1400, 1200, cfg)).toBe(3);
  });

  test('underdog wins 1200 vs 1400 → delta 29', () => {
    expect(calculateMatchDelta(1200, 1400, cfg)).toBe(29);
  });
});

describe('SECU-02: score_weighted preset — known outputs (must pass explicit scores)', () => {
  const base = { formulaPreset: 'score_weighted', kFactor: 32, dFactor: 200 };

  test('equal ratings, blowout 11-0 → delta 32', () => {
    expect(calculateMatchDelta(1200, 1200, { ...base, scoreWinner: 11, scoreLoser: 0 })).toBe(32);
  });

  test('equal ratings, close game 11-9 → delta 18', () => {
    expect(calculateMatchDelta(1200, 1200, { ...base, scoreWinner: 11, scoreLoser: 9 })).toBe(18);
  });

  test('no scores provided → falls through to standard ELO (delta 16 for equal ratings)', () => {
    // scoreWinner/scoreLoser null → the score_weighted conditional is skipped
    expect(calculateMatchDelta(1200, 1200, { formulaPreset: 'score_weighted', kFactor: 32, dFactor: 200 })).toBe(16);
  });
});

// ─── SECU-02: Edge cases ──────────────────────────────────────────────────────

describe('SECU-02: edge cases — numeric output not broken', () => {
  test('zero-score match (0-0) with score_weighted: falls through to standard (total=0 guard)', () => {
    const result = calculateMatchDelta(1200, 1200, {
      formulaPreset: 'score_weighted', kFactor: 32, dFactor: 200,
      scoreWinner: 0, scoreLoser: 0,
    });
    expect(typeof result).toBe('number');
    expect(isFinite(result)).toBe(true);
    // total=0 guard: effectiveK=kFactor, delta=16
    expect(result).toBe(16);
  });

  test('identical ELO ratings with standard preset → always returns 16 (kFactor=32)', () => {
    expect(calculateMatchDelta(800, 800, { formulaPreset: 'standard', kFactor: 32, dFactor: 200 })).toBe(16);
    expect(calculateMatchDelta(2400, 2400, { formulaPreset: 'standard', kFactor: 32, dFactor: 200 })).toBe(16);
  });

  test('custom formula with extreme constant returns finite number', () => {
    const result = calculateMatchDelta(1200, 1200, {
      formulaPreset: 'custom',
      customFormula: 'customC',
      customConstants: { customC: 999999 },
      kFactor: 32, dFactor: 200,
    });
    expect(typeof result).toBe('number');
    expect(isFinite(result)).toBe(true);
    expect(result).toBe(999999);
  });

  test('invalid custom formula → returns standard ELO fallback (finite number), logs to console.error', () => {
    const spy = jest.spyOn(console, 'error').mockImplementation(() => {});
    const result = calculateMatchDelta(1200, 1200, {
      formulaPreset: 'custom',
      customFormula: 'throw new Error("bad formula")',
      kFactor: 32, dFactor: 200,
    });
    expect(typeof result).toBe('number');
    expect(isFinite(result)).toBe(true);
    expect(result).toBe(16); // standard fallback for equal ratings
    expect(spy).toHaveBeenCalledWith(
      expect.stringContaining('[ELO]'),
      expect.objectContaining({ formula: 'throw new Error("bad formula")', error: expect.any(String) })
    );
    spy.mockRestore();
  });
});

// ─── validateCustomFormula ────────────────────────────────────────────────────

describe('SECU-02: validateCustomFormula — accept/reject boundary', () => {
  test('valid standard formula expression returns null', () => {
    expect(validateCustomFormula('Math.round(kFactor * (1 - expectedScore(winnerElo, loserElo)))')).toBeNull();
  });

  test('formula calling undefined function returns an error string', () => {
    const result = validateCustomFormula('not_a_function()');
    expect(typeof result).toBe('string');
    expect(result.length).toBeGreaterThan(0);
  });

  test('formula returning non-number returns an error string', () => {
    const result = validateCustomFormula('"not a number"');
    expect(typeof result).toBe('string');
    expect(result.length).toBeGreaterThan(0);
  });
});

// ─── SECU-01: FORMULA_PRESETS export available ────────────────────────────────

describe('SECU-01: FORMULA_PRESETS export — three presets accessible', () => {
  test('FORMULA_PRESETS has standard, score_weighted, custom keys', () => {
    expect(Object.keys(FORMULA_PRESETS)).toEqual(expect.arrayContaining(['standard', 'score_weighted', 'custom']));
  });

  test('each preset has template string', () => {
    for (const [, preset] of Object.entries(FORMULA_PRESETS)) {
      expect(typeof preset.template).toBe('string');
      expect(preset.template.length).toBeGreaterThan(0);
    }
  });
});
